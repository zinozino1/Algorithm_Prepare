import sys

input = sys.stdin.readline

cnt = 0
def dfs(L, check, n, m):
    global cnt
    if L == m:

        s = set(check)
        if len(s) <= m:
            return
        cnt += 1
        for i in range(m):
            print(check[i], end=" ")
        print()
    else:
        for i in range(n):
            check[L] = i + 1
            dfs(L + 1, check, n, m)

def sol():
    n, m = map(int, input().split())
    check = [0] * n
    dfs(0, check, n, m)
    print(cnt)


sol()
