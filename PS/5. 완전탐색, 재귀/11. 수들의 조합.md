# 내 풀이


import sys
input =  sys.stdin.readline

cnt = 0
tmp = []

def dfs(L, res, s,check,a,n,k,m):
  global cnt
  global tmp
  if L == k:
    if sum(res) % m == 0:
      cnt+=1

  else:
    for i in range(s,n):
      if check[i] == 0:
        check[i] = 1
        res[L] = a[i]
        dfs(L+1, res, i+1,check, a, n, k, m)
        check[i] = 0

def sol():
  n,k = map(int, input().split())
  a = list(map(int, input().split()))
  m = int(input())
  res = [0] * k
  check = [0] * n
  dfs(0, res, 0,check,a,n,k,m)
  print(cnt)

sol()
