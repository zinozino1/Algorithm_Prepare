input = sys.stdin.readline

minN = 1e9
cnt = 0


def dfs(L, sum, n, a, m):
    global minN
    global cnt
    if sum == m:
        if cnt < minN:
            minN = cnt
    if sum > m:
        return
    if cnt >= minN:
        return
    else:
        for i in range(len(a)):
            cnt += 1 -> L로 대체 가능
            sum += a[i] -> dfs함수 인자에 넣을 수 있음
            dfs(L + 1, sum, n, a, m)
            cnt -= 1
            sum -= a[i]

def sol():
    n = int(input())
    a = list(map(int, input().split()))
    m = int(input())
    a.sort(reverse=True) -> 그리디 개념을 섞자
    sum = 0
    dfs(0, sum, n, a, m)
    print(minN)


sol()
