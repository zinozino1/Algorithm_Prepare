def sol():
  global maxN
  c,n = map(int,input().split())
  target = []
  maxN = -1e9

  for _ in range(n):
    target.append(int(input()))

  totSum = sum(target)
  def dfs(L, tot, totS):
    global maxN

    if tot > c:
      return
    if totS+tot < maxN:
      return
    if L == n:
      if tot > maxN:
        maxN = tot
      return

    else:
    # totS-target[L] -> 해당 가지로부터 밑으로 남은 무게들
      dfs(L+1, tot+target[L], totS-target[L])
      dfs(L+1, tot, totS-target[L])

  dfs(0, 0, totSum)
  print(maxN)
