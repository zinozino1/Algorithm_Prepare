import time


# 약 1시간 20분 소요


def sol():
    a = [list(map(int, input().split())) for _ in range(9)]
    n = len(a)

    s = set()
    dx = [0, 1, 2, 0, 1, 2, 0, 1, 2]
    dy = [0, 0, 0, 1, 1, 1, 2, 2, 2]
    for i in range(0, n, 3):
        for j in range(0, n, 3):
            for k in range(9):
                s.add(a[i + dy[k]][j + dx[k]])
            if len(list(s)) != 9:
                print("NO")
                return
            s = set()
    print("YES")


start_time = int(round(time.time() * 1000))
sol()
end_time = int(round(time.time() * 1000))

print()
print('sol()의 실행 시간 : %d(ms)' % (end_time - start_time))

            ch1[a[i][j]]=1

            return False
    for i in range(3):
        for j in range(3):
            ch3=[0]*10
            for k in range(3):
                for s in range(3):
                    ch3[a[i*3+k][j*3+s]]=1
            if sum(ch3)!=9:
                return False
    return True

a=[list(map(int, input().split())) for _ in range(9)]
