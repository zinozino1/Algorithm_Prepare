# 내 풀이

import time

def sol():
    n = int(input())
    cnt = 0
    for i in range(2, n + 1):
        for j in range(1, i):
            if j == 1:
                continue
            if i % j == 0:
                break
        else:
            cnt += 1
    print(cnt)

start_time = int(round(time.time() * 1000))
sol()
end_time = int(round(time.time() * 1000))

print('sol()의 실행 시간 : %d(ms)' % (end_time - start_time))

# 답 - 에라토스테네스 체

import time


def sol():
    n = int(input())
    ch = [0] * (n + 1)  # 인덱스 자체가 값이기때문에 n + 1
    cnt = 0
    for i in range(2, n + 1):
        if ch[i] == 0:
            cnt += 1
            for j in range(i, n + 1, i): -> * step을 i 배수로 설정
                ch[j] = 1
    print(cnt)


start_time = int(round(time.time() * 1000))
sol()
end_time = int(round(time.time() * 1000))

print('sol()의 실행 시간 : %d(ms)' % (end_time - start_time))

-> 0.2초걸림
