# 내 풀이

def sol():
    n, m = map(int, input().split())
    res = []
    for i in range(1, n + 1):
        for j in range(1, m + 1):
            res.append(i + j)
    result = []
    for i in range(2, n + m + 1):
        cnt = 0
        for x in res:
            if i == x:
                cnt += 1
        result.append(cnt)
    maxNum = max(result)
    for i in range(len(result)):
        if result[i] == maxNum:
            print(i + 2, end=" ")


sol()
