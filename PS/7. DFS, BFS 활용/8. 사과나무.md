-> 다이아몬드, 모래시계모양은 lt,rt가 갑임

import sys
from collections import deque
input = sys.stdin.readline

def sol():
  n = int(input())
  board = []
  for _ in range(n):
    board.append(list(map(int,input().split())))

  mid = n // 2
  lt = 0
  rt = 0
  sum = 0

  for i in range(n):
    for j in range(mid-lt, mid+rt+1):
      sum += board[i][j]
    if i < n // 2:
      lt += 1
      rt += 1
    else:
      lt -= 1
      rt -= 1
  print(sum)



sol()

import sys
from collections import deque
input = sys.stdin.readline

dx = [-1,0,1,0]
dy = [0,1,0,-1]


def sol():
  n = int(input())
  board = []
  for _ in range(n):
    board.append(list(map(int,input().split())))
  check = [[0] * n for _ in range(n)]
  q = deque()
  q.append((n//2,n//2))

  L=0
  sum = board[n//2][n//2]
  check[n//2][n//2] = 1

  while True: # 레벨마다 순회해줘야함
    if L == n//2: # 종료 조건
      break

    for i in range(len(q)): # 가지 뻗어나가는 부분 -> 큐의 길이만큼 반복문 돌려줘야함 ****** 중요
      curr = q.popleft()
      for next in [(curr[0]+dx[s], curr[1]+dy[s]) for s in range(4)]:
        if check[next[0]][next[1]] == 0:
          check[next[0]][next[1]] = 1
          q.append(next)
          sum += board[next[0]][next[1]]
    L += 1

  print(sum)

sol()
