# 내 풀이

import sys
input = sys.stdin.readline

def sol():
  global acnt,bcnt,ccnt,minN
  n = int(input())
  a = [int(input()) for _ in range(n)]
  acnt,bcnt,ccnt = 0,0,0
  minN = 1e9

  def dfs(L, atot, btot, ctot):
    global acnt,bcnt,ccnt,minN

    # if max(atot,btot,ctot)-min(atot,btot,ctot) > minN:
    #   return

    if acnt > n-2 or bcnt > n-2 or ccnt > n-2:
      return
    if L == n:
      if acnt == 0 or bcnt == 0 or ccnt == 0:
        return
      if atot == btot or btot == ctot or atot == ctot:
        return
      #print("[",atot, btot, ctot,"]")
      if max(atot,btot,ctot)-min(atot,btot,ctot) < minN:
        minN = max(atot,btot,ctot)-min(atot,btot,ctot)

    else:
      acnt+=1
      dfs(L+1, atot+a[L],btot,ctot)
      acnt-=1
      bcnt+=1
      dfs(L+1, atot,btot+a[L],ctot)
      bcnt-=1
      ccnt+=1
      dfs(L+1, atot,btot,ctot+a[L])
      ccnt-=1


  dfs(0,0,0,0)
  print(minN)

sol()
