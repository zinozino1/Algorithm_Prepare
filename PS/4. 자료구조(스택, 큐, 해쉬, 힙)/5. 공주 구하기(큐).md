import time
import bisect
import sys
from collections import deque


def sol():
    n, k = map(int, input().split())
    q = deque(list(range(1, n + 1)))
    num = 0
    while True:
        if len(q) == 1:
            break
        tmp = q.popleft()
        num += 1
        if num == k:
            num = 0
        else:
            q.append(tmp)
    print(q.pop())


start_time = int(round(time.time() * 1000))
sol()
end_time = int(round(time.time() * 1000))

print()
print('sol()의 실행 시간 : %d(ms)' % (end_time - start_time))
