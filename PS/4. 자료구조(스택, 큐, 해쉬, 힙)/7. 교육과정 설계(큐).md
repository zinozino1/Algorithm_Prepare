* 내장 함수 연습 좀 해야할듯

import time
import bisect
import sys
from collections import deque


def sol():
    order = input()
    n = int(input())
    arr = [list(input()) for _ in range(n)]

    for i in range(n):
        res = ''
        q = deque(arr[i])

        while q:
            curr = q.popleft()
            if curr in order: * 이런거 문법 좀 파악해라
                if curr not in res:
                    res += curr
        if res == order:
            print("#%d YES" % (i + 1))
        else:
            print("#%d NO" % (i + 1))



start_time = int(round(time.time() * 1000))
sol()
end_time = int(round(time.time() * 1000))

print()
print('sol()의 실행 시간 : %d(ms)' % (end_time - start_time))
