def sol():
    n = int(input())
    d = dict()
    for i in range(n):
        d[input()] = 1 # 해쉬는 같은 값이 들어올 경우 덮어쓰는 특징이 있음
    for i in range(n-1):
        d[input()] = 2
    arr = list(d.items())
    for x in arr:
        if x[1] == 1:
            print(x[0])

def sol():
  n = int(input())

  dic = dict()

  for i in range(n):
    curr = input().strip()
    dic[curr] = 0
  for i in range(n-1):
    curr = input().strip()
    dic[curr] = 1

  for x in dic.keys():
    if dic[x] == 0:
      print(x)
