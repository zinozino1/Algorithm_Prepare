# 최대부분 증가수열과 논리 같음

import sys
input = sys.stdin.readline

def sol():
  n = int(input())
  arr = list(map(int, input().split()))

  dp = [0] * n
  dp[0] = 1

  for i in range(1,n):
    maxN = 0
    for j in range(i-1, -1,-1):
      if arr[i] > arr[j]:
        if dp[j] > maxN:
          maxN = dp[j]
    dp[i] = maxN + 1
  print(max(dp))

sol()

def sol():
  n = int(input())
  target = list(map(int,input().split()))
  dp = [0] * n
  dp[0] = 1

  for i in range(1, n):
    tmp = []
    for j in range(i-1, -1, -1):
      if target[j] < target[i]:
        tmp.append(dp[j])
    if tmp:
      dp[i] = max(tmp) + 1
    else:
      dp[i] = 1

  print(max(dp))

sol()
