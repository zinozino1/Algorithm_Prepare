# 내 풀이


import sys
input = sys.stdin.readline
from collections import deque

dx = [-1,0,1,0,1,-1,1,-1]
dy = [0,1,0,-1,1,1,-1,-1]

def sol():
  n = int(input())
  land = [list(map(int, input().split())) for _ in range(n)]

  dis = [[0] * n for _ in range(n)]
  q = deque()

  cnt = 0

  for i in range(n):
    for j in range(n):
      if land[i][j] == 1 and dis[i][j] == 0:
        q.append((i,j))
        dis[i][j] = 1
        while q:
          for k in range(len(q)):
            curr = q.popleft()
            dis[curr[0]][curr[1]] = 1
            for s in range(8):
              nx = curr[0]+dx[s]
              ny = curr[1]+dy[s]

              if 0<=nx<=n-1 and 0<=ny<=n-1 and dis[nx][ny] == 0 and land[nx][ny] == 1:
                q.append((nx,ny))
        cnt += 1

  print(cnt)

sol()

def sol():

  n = int(input())
  land = [list(map(int,input().split())) for _ in range(n)]

  cnt = 0
  q = deque()
  check = [[0] * n for _ in range(n)]
  for i in range(n):
    for j in range(n):
      if land[i][j] == 1:
        q.append((i,j))
        check[i][j] = 1
        cnt += 1

        while q:
          for k in range(len(q)):
            curr = q.popleft()

            for s in range(8):
              nx = curr[0]+dx[s]
              ny = curr[1]+dy[s]
              if 0<=nx<=n-1 and 0<=ny<=n-1 and check[nx][ny] == 0 and land[nx][ny] == 1:
                q.append((nx,ny))
                land[nx][ny] = -1 # 이거 무조건 해줘야 돼
                check[nx][ny] = 1

  print(cnt)

sol()
