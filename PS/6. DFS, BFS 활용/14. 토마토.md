# 내 풀이
-> 핵심 아이디어는 bfs의 시작점이 '익은 토마토들'이라는 것


import sys
from collections import deque
input = sys.stdin.readline

dx = [-1,0,1,0]
dy = [0,1,0,-1]

def sol():
  m,n = map(int,input().split())
  box = [list(map(int, input().split())) for _ in range(n)]
  check = [[0]*m for _ in range(n)]
  maxN = -1e9
  flag = False

  q = deque()

  for i in range(n):
    for j in range(m):
      if box[i][j] == 1:
        q.append((i,j))
        check[i][j] = 1

  cnt = 1
  while q:
    for i in range(len(q)):
      curr = q.popleft()
      for s in range(4):
        nx = curr[0]+dx[s]
        ny = curr[1]+dy[s]
        if 0<=nx<=n-1 and 0<=ny<=m-1 and check[nx][ny] == 0 and box[nx][ny] == 0:
          q.append((nx,ny))
          box[nx][ny] = cnt
          check[nx][ny] = 1
    cnt += 1

  for x in box:
    if max(x) > maxN:
      maxN = max(x)
  for x in box:
    if 0 in x:
      flag = True
  if flag:
    print(-1)
  else:
    print(maxN)

sol()
