# 내 풀이

-> 답은 맞았으나 시간초과날 것 같음
-> 거리를 계산할 때 굳이 BFS를 사용할 이유가 없었다

import sys
from collections import deque
input = sys.stdin.readline

dx = [-1,0,1,0]
dy = [0,1,0,-1]

def sol():
  global minCnt
  n,m = map(int,input().split())
  city = [list(map(int,input().split())) for _ in range(n)]
  tmpCity = [[0]*n for _ in range(n)]
  pizza = []
  minCnt = 1e9
  for i in range(len(city)):
    for j in range(len(city[i])):
      if city[i][j] == 2:
        pizza.append((i,j))
      if city[i][j] == 1:
        tmpCity[i][j] = 1

  def dfs(L,s,tmpCity):
    global minCnt
    if L == m:
      sum = 0
      for i in range(n):
        for j in range(n):
          if tmpCity[i][j] == 1:
            q = deque()
            q.append((i,j))
            check = [[0]*n for _ in range(n)]
            check[i][j] = 1
            cnt = 0
            while q:
              cnt += 1
              for k in range(len(q)):
                curr = q.popleft()
                if tmpCity[curr[0]][curr[1]] == 2:
                  q.clear()
                  break
                for s in range(4):
                  nx = curr[0]+dx[s]
                  ny = curr[1]+dy[s]
                  if 0<=nx<=n-1 and 0<=ny<=n-1 and check[nx][ny] == 0:
                    check[nx][ny] = 1
                    q.append((nx,ny))
            sum += cnt - 1
      if sum < minCnt:
        minCnt = sum

    else:
      for i in range(s,len(pizza)):
        tmpCity[pizza[i][0]][pizza[i][1]] = 2
        dfs(L+1, i+1, tmpCity)
        tmpCity[pizza[i][0]][pizza[i][1]] = 0

  dfs(0,0,tmpCity)
  print(minCnt)

sol()
