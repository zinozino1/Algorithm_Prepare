    a,b = map(int,input().split())
    table.append((i+1,a,b))

    global maxN
    if tot > maxN:
      maxN = tot
    for i in range(s, n):
      if s == 0:
        dfs(i+1, tot+table[i][2])
      elif table[s-1][0] + table[s-1][1] == table[i][0]:
        dfs(i+1, tot+table[i][2])

  dfs(0, 0)

  print(maxN)

  for i in range(n):
    T,P = map(int,input().split())
    table.append((i+1,T,P))
  maxN = -1e9

  def dfs(L, tot):
    global maxN
    if L == n:
      if tot > maxN:
        maxN = tot
    else:
      if table[L][0] + table[L][1] <= n+1:
        dfs(table[L][0]+table[L][1]-1,tot+table[L][2])
      dfs(L+1,tot)
  dfs(0,0)
  print(maxN)

# 2차 풀이

import sys

input = sys.stdin.readline

def sol():
  global maxN
  n = int(input())
  schedule = []
  maxN = -1e9

  for _ in range(n):
    T,P = map(int, input().split())
    schedule.append((T,P))

  def dfs(L,time,tot):
    global maxN
    if L >= n:
      if tot > maxN and L == n:
        maxN = tot
      return
    else:
      dfs(L+schedule[L][0], time+schedule[L][0], tot+schedule[L][1])
      dfs(L+1, time, tot)

  dfs(0,0,0)
  print(maxN)

sol()
