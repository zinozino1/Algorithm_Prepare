# 내 풀이
-> BFS로 풀긴 했는데 약간 헤맴

import sys
input = sys.stdin.readline
from collections import deque

dx = [-1,0,1,0]
dy = [0,1,0,-1]

def sol():
  n = int(input())
  board = [list(input().strip()) for _ in range(n)]
  dis = [[0]*n for _ in range(n)]

  cnt = 1

  for i in range(n):
    for j in range(n):
      if board[i][j] == "1" and dis[i][j] == 0:
        q = deque()
        q.append((i,j))
        # tmpCnt = 0
        while q: -> 한 단지 체킹
          for k in range(len(q)):
            curr = q.popleft()
            dis[curr[0]][curr[1]] = cnt

            for s in range(4):
              nx = curr[0]+dx[s]
              ny = curr[1]+dy[s]
              if 0<=nx<=n-1 and 0<=ny<=n-1 and dis[nx][ny] == 0 and board[nx][ny] == "1":
                q.append((nx,ny))
                #tmpCnt += 1
        #print(tmpCnt)
        cnt+=1

  print(cnt-1)
  res = []
  for i in range(1,cnt):
    cnt = 0
    for x in dis:
      cnt+=x.count(i)
    res.append(cnt)
  res.sort()
  for x in res:
    print(x)







sol()
