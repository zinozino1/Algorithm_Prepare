import sys
input = sys.stdin.readline


def sol():
  global cnt
  s = input() # 문자열을 배열로 취급하면 마지막이 '/n'인것을 주의
  cnt = 0

  def dfs(L, ch, s):
    global cnt
    if len(s) == 1:
      print(ch)
      cnt+=1
      return
    else:
      if len(s) > 1 and s[:1] != '0':
        # 1칸 자르는 경우
        dfs(L+1, ch+chr(int(s[:1])+64), s[1:len(s)])
        if int(s[:2])+64 < 90:
          # 2칸 자르는 경우
          dfs(L+1, ch+chr(int(s[:2])+64), s[2:len(s)])

  dfs(0,'',s)
  print(cnt)

sol()
