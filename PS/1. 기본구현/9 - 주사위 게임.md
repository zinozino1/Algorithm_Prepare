import time

# 40여분 소요

def part1(x):  # 1차원 배열 받음
    # newArr = list(set(x))
    cnt = 0
    com = -1
    for i in range(len(x)):
        for j in range(i + 1, len(x)):
            if x[i] == x[j]:
                cnt += 1
                com = x[i]
    return cnt, com


def sol():
    n = int(input())
    a = [list(map(int, input().split())) for _ in range(n)]
    score = 0
    maxNum = -1e9
    for x in a:
        cnt, com = part1(x)
        if cnt == 0:
            score = max(x) * 100
        elif cnt == 1:
            score = com * 100 + 1000
        else:
            score = com * 1000 + 10000
        if maxNum < score:
            maxNum = score
    print(maxNum)


start_time = int(round(time.time() * 1000))
sol()
end_time = int(round(time.time() * 1000))

print()
print('sol()의 실행 시간 : %d(ms)' % (end_time - start_time))
