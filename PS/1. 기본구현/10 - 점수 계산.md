def sol():

  n = int(input())

  target = list(map(int, input().split()))
  res = [0] * n
  cnt = 0
  for i in range(n):
    if target[i] == 0:
      cnt = 0
      res[i] = 0
    elif target[i] == 1:
      cnt += 1
      if cnt > 0:
        res[i] = cnt

  print(sum(res))
